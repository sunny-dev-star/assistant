## 一、总体策略：以"可交付"为最高优先级

当前最大的风险是：**基础设施完备，但没有一个完整的端到端流程可以交付给真实客户**。

落地原则：

1. **纵向打通优先于横向铺开** — 先让一个租户 → 一个技能 → 一个渠道完整跑通，再扩展
2. **多租户是一切的前提** — 当前测试脚本里没有 `tenant_id`，这是最紧迫的修复点
3. **每个阶段结束必须有可演示的产物** — 不做没有外部可见结果的内部重构

```python
# src/assistant/interfaces/middleware/tenant_auth.py
from fastapi import Request, HTTPException
from starlette.middleware.base import BaseHTTPMiddleware
from assistant.infrastructure.repositories.tenant_repo import TenantRepository

class TenantAuthMiddleware(BaseHTTPMiddleware):
    """
    从请求头 Authorization: Bearer {api_key} 中解析租户身份。
    将 tenant 对象注入 request.state.tenant，供后续所有层使用。
    """
    def __init__(self, app, tenant_repo: TenantRepository):
        super().__init__(app)
        self.tenant_repo = tenant_repo

    async def dispatch(self, request: Request, call_next):
        # 跳过公开端点
        if request.url.path in ["/health", "/docs", "/openapi.json"]:
            return await call_next(request)

        api_key = self._extract_api_key(request)
        if not api_key:
            raise HTTPException(status_code=401, detail="Missing API Key")

        tenant = await self.tenant_repo.get_by_api_key(api_key)
        if not tenant or tenant.status != "active":
            raise HTTPException(status_code=401, detail="Invalid or inactive tenant")

        # 配额检查
        if tenant.quota_used >= tenant.quota_limit:
            raise HTTPException(status_code=429, detail="Quota exceeded")

        request.state.tenant = tenant
        response = await call_next(request)
        return response

    def _extract_api_key(self, request: Request) -> str | None:
        auth_header = request.headers.get("Authorization", "")
        if auth_header.startswith("Bearer "):
            return auth_header[7:]
        return None
```

```python
# src/assistant/infrastructure/repositories/tenant_repo.py
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from assistant.domain.entities.tenant import Tenant

class TenantRepository:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def get_by_api_key(self, api_key: str) -> Tenant | None:
        result = await self.session.execute(
            select(Tenant).where(Tenant.api_key == api_key)
        )
        return result.scalar_one_or_none()

    async def get_by_id(self, tenant_id: str) -> Tenant | None:
        result = await self.session.execute(
            select(Tenant).where(Tenant.id == tenant_id)
        )
        return result.scalar_one_or_none()

    async def increment_quota(self, tenant_id: str, tokens: int):
        tenant = await self.get_by_id(tenant_id)
        if tenant:
            tenant.quota_used += tokens
            await self.session.commit()

    async def create(self, tenant: Tenant) -> Tenant:
        self.session.add(tenant)
        await self.session.commit()
        await self.session.refresh(tenant)
        return tenant
```

### Task 1.3：实现上下文管理服务（ConversationContextService）

**文件**: `src/assistant/domain/services/conversation_context_service.py`（补全）

```python
# src/assistant/domain/services/conversation_context_service.py
from typing import List
from assistant.domain.entities.message import Message
from assistant.infrastructure.repositories.message_repo import MessageRepository

class ConversationContextService:
    """
    管理对话上下文窗口。
    - 默认保留最近 10 轮（可按租户配置，最大 50 轮）
    - 超长时自动摘要（调用 LLM 生成摘要替换早期消息）
    """
    DEFAULT_WINDOW = 10
    MAX_WINDOW = 50

    def __init__(self, message_repo: MessageRepository, llm_port):
        self.message_repo = message_repo
        self.llm_port = llm_port

    async def get_context(
        self,
        conversation_id: str,
        window_size: int = DEFAULT_WINDOW
    ) -> List[dict]:
        """获取格式化后的上下文消息列表，直接传给 LLM"""
        window_size = min(window_size, self.MAX_WINDOW)
        messages = await self.message_repo.get_recent(
            conversation_id, limit=window_size * 2  # user+assistant 各算一条
        )
        return [{"role": m.role, "content": m.content} for m in messages]

    async def get_context_with_summary(
        self,
        conversation_id: str,
        window_size: int = DEFAULT_WINDOW
    ) -> List[dict]:
        """
        超过窗口时，将早期消息压缩为摘要。
        摘要作为 system 消息插入上下文头部。
        """
        all_messages = await self.message_repo.get_all(conversation_id)
        if len(all_messages) <= window_size * 2:
            return [{"role": m.role, "content": m.content} for m in all_messages]

        # 分割：早期消息 → 摘要；近期消息 → 保留
        older = all_messages[: -window_size * 2]
        recent = all_messages[-window_size * 2 :]

        summary = await self._summarize(older)
        context = [{"role": "system", "content": f"[对话摘要] {summary}"}]
        context += [{"role": m.role, "content": m.content} for m in recent]
        return context

    async def _summarize(self, messages: List[Message]) -> str:
        text = "\n".join([f"{m.role}: {m.content}" for m in messages])
        prompt = f"请将以下对话压缩为一段简短摘要，保留关键信息：\n\n{text}"
        response = await self.llm_port.complete([
            {"role": "user", "content": prompt}
        ])
        return response.content
```

```python
# src/assistant/interfaces/routers/chat_router.py
from fastapi import APIRouter, Request, Depends
from pydantic import BaseModel
from assistant.application.services.assistant_chat_app_service import AssistantChatAppService

router = APIRouter(prefix="/v1")

class ChatRequest(BaseModel):
    message: str
    session_id: str | None = None   # 不传则自动生成
    user_id: str
    channel: str = "api"            # api / wechat / feishu / dingtalk

class ChatResponse(BaseModel):
    reply: str
    session_id: str
    tokens_used: int

@router.post("/chat", response_model=ChatResponse)
async def chat(
    body: ChatRequest,
    request: Request,
    app_service: AssistantChatAppService = Depends()
):
    """
    统一对话端点。tenant 由中间件从 Bearer Token 注入 request.state.tenant。
    """
    tenant = request.state.tenant
    result = await app_service.execute(
        tenant=tenant,
        message=body.message,
        user_id=body.user_id,
        session_id=body.session_id,
        channel=body.channel,
    )
    return ChatResponse(
        reply=result.reply,
        session_id=result.session_id,
        tokens_used=result.tokens_used,
    )
```

```python
# src/assistant/interfaces/routers/wechat_router.py
from fastapi import APIRouter, Request, Response, Query
from assistant.infrastructure.adapters.wechat_adapter import WechatAdapter

router = APIRouter(prefix="/webhook")

@router.get("/wechat/{tenant_id}")
async def wechat_verify(
    tenant_id: str,
    signature: str = Query(...),
    timestamp: str = Query(...),
    nonce: str = Query(...),
    echostr: str = Query(...)
):
    """微信服务器验证"""
    adapter = WechatAdapter.for_tenant(tenant_id)
    if adapter.verify_signature(signature, timestamp, nonce):
        return Response(content=echostr, media_type="text/plain")
    return Response(status_code=403)

@router.post("/wechat/{tenant_id}")
async def wechat_message(tenant_id: str, request: Request):
    """接收微信消息，路由到 AppService，返回 XML"""
    adapter = WechatAdapter.for_tenant(tenant_id)
    msg = await adapter.parse_message(request)

    if msg.msg_type == "voice":
        # 语音消息：先转文字
        msg.content = await adapter.transcribe_voice(msg.media_id)

    # 路由到 AppService（此处复用 /v1/chat 的 AppService）
    app_service = request.app.state.chat_app_service
    result = await app_service.execute(
        tenant_id=tenant_id,
        message=msg.content,
        user_id=msg.from_user,
        channel="wechat"
    )

    return Response(
        content=adapter.build_text_reply(msg.from_user, msg.to_user, result.reply),
        media_type="application/xml"
    )
```

```python
async def transcribe_voice(self, media_id: str) -> str:
    """
    微信语音消息处理：
    1. 调用微信 API 下载 .amr 文件
    2. 调用 iFlytek / 阿里云 ASR 转文字
    3. 返回文字内容
    """
    # Step 1: 获取微信 media 文件
    access_token = await self.get_access_token()
    url = f"https://api.weixin.qq.com/cgi-bin/media/get?access_token={access_token}&media_id={media_id}"
    async with httpx.AsyncClient() as client:
        response = await client.get(url)
        audio_bytes = response.content

    # Step 2: 调用 ASR（以阿里云为例）
    asr_client = AliyunASRClient(self.config["asr_key"])
    text = await asr_client.transcribe(audio_bytes, format="amr")
    return text
```

### Task 2.3：完成 SkillLoader 实现（补全 DESIGN.md 中的 stub）

```python
# src/assistant/infrastructure/skill_loader.py
import yaml
import re
from pathlib import Path
from typing import Dict, List, Optional

class SkillLoader:
    """
    按 Claude Skill 规范加载技能包。
    三级加载：元数据 → 主体说明 → 按需脚本/知识
    """

    def __init__(self, skills_dir: str = "skills"):
        self.skills_dir = Path(skills_dir)
        self._cache: Dict[str, dict] = {}
        self.load_all()

    def load_all(self):
        for skill_dir in self.skills_dir.iterdir():
            if skill_dir.is_dir() and (skill_dir / "SKILL.md").exists():
                skill = self._load_skill(skill_dir)
                if skill:
                    self._cache[skill["name"]] = skill

    def _load_skill(self, skill_dir: Path) -> Optional[dict]:
        content = (skill_dir / "SKILL.md").read_text(encoding="utf-8")

        # 解析 Frontmatter
        match = re.match(r"^---\n(.*?)\n---\n(.*)", content, re.DOTALL)
        if not match:
            return None

        frontmatter = yaml.safe_load(match.group(1))
        body = match.group(2).strip()

        # 加载 references/（拼入 System Prompt）
        references = []
        refs_dir = skill_dir / "references"
        if refs_dir.exists():
            for ref_file in sorted(refs_dir.glob("*.md")):
                references.append(ref_file.read_text(encoding="utf-8"))

        return {
            "name": frontmatter.get("name"),
            "description": frontmatter.get("description"),
            "version": frontmatter.get("version", "1.0.0"),
            "tools": frontmatter.get("tools", []),       # LLM tool_calls 的 schema
            "system_prompt": "\n\n".join(references),   # 注入 System Prompt
            "body": body,
            "scripts_dir": skill_dir / "scripts",
        }

    def get(self, skill_name: str) -> Optional[dict]:
        return self._cache.get(skill_name)

    def get_tools_schema(self, skill_name: str) -> List[dict]:
        skill = self.get(skill_name)
        return skill["tools"] if skill else []

    def list_skills(self) -> List[str]:
        return list(self._cache.keys())

    async def execute_tool(self, skill_name: str, tool_name: str, params: dict) -> str:
        """动态执行 scripts/ 下对应工具脚本"""
        import importlib.util, json, sys
        skill = self.get(skill_name)
        if not skill:
            return f"技能 {skill_name} 未找到"

        script_path = skill["scripts_dir"] / f"{tool_name}.py"
        if not script_path.exists():
            return f"工具 {tool_name} 未实现"

        spec = importlib.util.spec_from_file_location(tool_name, script_path)
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)

        func = getattr(module, tool_name, None)
        if func:
            result = await func(**params) if asyncio.iscoroutinefunction(func) else func(**params)
            return json.dumps(result, ensure_ascii=False)
        return "工具执行失败"
```

```python
import httpx
from enum import Enum

class AlertLevel(Enum):
    EMERGENCY = "🚨 紧急"
    WARNING   = "⚠️ 警告"
    INFO      = "ℹ️ 通知"

class FeishuAlerter:
    def __init__(self, webhook_url: str):
        self.webhook_url = webhook_url

    async def send(self, level: AlertLevel, title: str, detail: str):
        payload = {
            "msg_type": "interactive",
            "card": {
                "header": {
                    "title": {"content": f"{level.value} {title}", "tag": "plain_text"},
                    "template": "red" if level == AlertLevel.EMERGENCY else "yellow"
                },
                "elements": [{
                    "tag": "div",
                    "text": {"content": detail, "tag": "lark_md"}
                }]
            }
        }
        async with httpx.AsyncClient() as client:
            await client.post(self.webhook_url, json=payload)
```

```python
from apscheduler.schedulers.asyncio import AsyncIOScheduler

scheduler = AsyncIOScheduler()

async def setup_tenant_jobs(tenant_id: str, config: dict):
    """为租户配置定时主动关怀任务"""

    # 每日早上 8:30 问候

        replace_existing=True
    )
```
